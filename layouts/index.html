{{ define "main" }} {{- $leaves := site.Data.book.leaves -}} {{/* ── Build spreads at build time ────────────────────────── Walk the leaves once, pairing adjacent compatible ones into spreads. Solo kinds (cover/title/end) never pair; image+image with a matching `pair` field becomes a stitched scene. */}} {{- $spreads := slice -}} {{- $skip := false -}} {{- range $i, $leaf := $leaves -}} {{- if $skip -}} {{- $skip = false -}} {{- else -}} {{- $kind := $leaf.kind -}} {{- $isSolo := or (eq $kind "cover") (or (eq $kind "title") (eq $kind "end")) -}} {{- $next := false -}} {{- $nextKind := "" -}} {{- if lt (add $i 1) (len $leaves) -}} {{- $next = index $leaves (add $i 1) -}} {{- $nextKind = $next.kind -}} {{- end -}} {{- $canPair := and (and (not $isSolo) (ne $next false)) (or (eq $nextKind "verse") (eq $nextKind "image")) -}} {{- if $canPair -}} {{- $type := "image-verse" -}} {{- if and (eq $kind "image") (eq $nextKind "image") -}} {{- $pairA := "" -}}{{- with $leaf.pair }}{{ $pairA = . }}{{ end -}} {{- $pairB := "" -}}{{- with $next.pair }}{{ $pairB = . }}{{ end -}} {{- if and (ne $pairA "") (eq $pairA $pairB) -}} {{- $type = "stitched-image" -}} {{- else -}} {{- $type = "double-image" -}} {{- end -}} {{- else if and (eq $kind "verse") (eq $nextKind "verse") -}} {{- $type = "double-verse" -}} {{- end -}} {{- $spreads = $spreads | append (dict "type" $type "leaves" (slice $leaf $next) "startIndex" $i) -}} {{- $skip = true -}} {{- else -}} {{- $spreads = $spreads | append (dict "type" (printf "solo-%s" $kind) "leaves" (slice $leaf) "startIndex" $i) -}} {{- end -}} {{- end -}} {{- end -}} {{/* Reading-time estimate from verse word count, ~180wpm. */}} {{- $words := 0 -}} {{- range $leaves -}} {{- if eq .kind "verse" -}} {{- $words = add $words (len (split .text " ")) -}} {{- end -}} {{- end -}} {{- $minutes := math.Round (div (mul $words 1.0) 180.0) -}} {{- if lt $minutes 1 -}}{{- $minutes = 1 -}}{{- end -}}
{{- range $sIdx, $spread := $spreads -}}
{{- range $li, $leaf := $spread.leaves -}} {{- partial (printf "leaf-%s.html" $leaf.kind) (dict "leaf" $leaf "index" (add $spread.startIndex $li) "minutes" $minutes "spreadCount" (len $spreads)) -}} {{- end -}}
{{- end -}}
1 of {{ len $spreads }}

{{ end }}
